## ðŸ”§ Device Selection Algorithm

```python
# Smart device selection based on signal strength matrix
SIGNAL_MATRIX = {
    "Dharmani Admin": {"office": -13, "admin": None, "router": None, "lab": None},
    "Airtel_Dharmani": {"office": -14, "admin": None, "router": None, "lab": None},
    "Dharmani Office": {"office": -80, "admin": -40, "router": -60, "lab": -68},
    "Dharmani Guest": {"office": -52, "admin": -41, "router": -60, "lab": -70},
    "Airtel_vish_0615": {"office": None, "admin": -71, "router": -45, "lab": -66},
    "OpenWrt": {"office": None, "admin": None, "router": None, "lab": -30},
    "OpenWrt+": {"office": None, "admin": None, "router": None, "lab": -20},
}

def select_best_device(target_network):
    """Select device with strongest signal for target network"""
    signals = SIGNAL_MATRIX.get(target_network, {})

    # Filter out None values (network not visible)
    valid_signals = {dev: sig for dev, sig in signals.items() if sig is not None}

    if not valid_signals:
        return None, None, "Network not visible from any device"

    # Get device with strongest signal (most negative dBm)
    best_device = min(valid_signals, key=valid_signals.get)
    best_signal = valid_signals[best_device]

    # Assess success probability
    if best_signal >= -30:
        success_rate = "99%+"
        quality = "EXCEPTIONAL"
    elif best_signal >= -50:
        success_rate = "95%+"
        quality = "EXCELLENT"
    elif best_signal >= -60:
        success_rate = "85%+"
        quality = "GOOD"
    elif best_signal >= -70:
        success_rate = "60%+"
        quality = "FAIR"
    else:
        success_rate = "<30%"
        quality = "WEAK"

    return best_device, best_signal, f"{quality} - {success_rate} expected success"

# Example usage:
device, signal, assessment = select_best_device("Dharmani Admin")
# Returns: ("office", -13, "EXCEPTIONAL - 99%+ expected success")
```
